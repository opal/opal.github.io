def self.dump(obj, io = nil, limit = nil)
  string = generate(obj)

  if io
    io = io.to_io if io.responds_to? :to_io
    io.write string

    io
  else
    string
  end
end

def self.from_object(js_object, options = {})
  options[:object_class] ||= Hash
  options[:array_class]  ||= Array

  `to_opal(js_object, options.smap)`
end
