def on_if(node)
  test, true_body, false_body = *node.children

  if skip_check_present?(test)
    false_body = s(:nil)
  end

  if skip_check_present_not?(test)
    true_body = s(:nil)
  end

  node.updated(nil, process_all([test, true_body, false_body]))
end
