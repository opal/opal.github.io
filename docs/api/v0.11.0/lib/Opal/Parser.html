def default_parser
  parser = super

  parser.diagnostics.all_errors_are_fatal = true
  parser.diagnostics.ignore_warnings      = false

  if RUBY_ENGINE == 'opal'
    parser.diagnostics.consumer = ->(diag){}
  else
    parser.diagnostics.consumer = diagnostics_consumer
  end

  parser
end
