def initialize(size, bits = 8)
  if native?(size)
    super(size)
  else
    super(`new ArrayBuffer(size * (bits / 8))`)
  end
end

def self.name_for(bits, type)
  part = case type
         when :unsigned then 'Uint'
         when :signed   then 'Int'
         when :float    then 'Float'
         end
  "#{part}#{bits}"
end
