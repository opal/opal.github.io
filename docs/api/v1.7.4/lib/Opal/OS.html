def bash_c(*commands)
  cmd = if windows?
          [
            'bundle',
            'exec',
            'cmd',
            '/c',
          ]
        else
          [
            'bundle',
            'exec',
            'bash',
            '-c',
          ]
        end

  cmd << commands.join(cmd_sep)
end

def shellescape(str)
  if windows?
    '"' + str.gsub('"', '""') + '"'
  else
    str.shellescape
  end
end
