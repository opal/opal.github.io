def process(node)
  return if node.nil?

  @_on_handler_cache ||= {}
  type = node.type

  on_handler = @_on_handler_cache[type] ||= begin
    handler = :"on_#{type}"
    handler = :handler_missing unless respond_to?(handler)
    handler
  end

  send(on_handler, node) || node
end
