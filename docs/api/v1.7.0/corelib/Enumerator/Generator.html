def each(*args, &block)
  yielder = Yielder.new(&block)

  %x{
    try {
      args.unshift(#{yielder});

      Opal.yieldX(#{@block}, args);
    }
    catch (e) {
      if (e && e.$thrower_type == "breaker") {
        return e.$v;
      }
      else {
        throw e;
      }
    }
  }

  self
end
