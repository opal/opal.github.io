def try
  if @wait.all?(&:realized?)
    promise = @wait.find(&:rejected?)
    if promise
      reject(promise.error)
    else
      resolve(@wait.map(&:value))
    end
  end
end

def wait(promise)
  unless Promise === promise
    promise = Promise.value(promise)
  end

  if promise.act?
    promise = promise.then
  end

  @wait << promise

  promise.always do
    try if @next.any?
  end

  self
end
