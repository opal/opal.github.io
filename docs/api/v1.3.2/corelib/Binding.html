def js_eval(*args)
  if @jseval
    @jseval.call(*args)
  else
    raise 'Evaluation on a Proc#binding is not supported'
  end
end

def local_variable_get(symbol)
  js_eval(symbol)
rescue Exception
  raise NameError, "local variable `#{symbol}' is not defined for #{inspect}"
end
