def initialize(buffer, bits = nil, type = nil)
  if Native == buffer
    super(buffer)
  else
    %x{
      var klass = #{Array.for(bits, type)};

      #{super(`new klass(#{buffer.to_n})`)}
    }
  end

  @buffer = buffer
  @type   = type
end

def each
  return enum_for :each unless block_given?

  %x{
    for (var i = 0, length = #{@native}.length; i < length; i++) {
      #{yield `#{@native}[i]`}
    }
  }

  self
end
