def source_map
  @source_map ||= begin
    manual_fragments = source.each_line.with_index.map do |line_source, index|
      column = line_source.index(/\S/)
      line = index + 1
      ManualFragment.new(line, column, line_source, nil)
    end

    ::Opal::SourceMap::File.new(manual_fragments, filename, source)
  end
end
