def compiled
  @compiled ||= begin
    compiler = compiler_for(@source, file: @filename.gsub(/\.(rb|js|opal)#{REGEXP_END}/, ''))
    compiler.compile
    compiler
  end
end

def required_trees
  compiled.required_trees.map do |tree|
    # Remove any leading ./ after joining to dirname
    File.join(File.dirname(@filename), tree).sub(%r{^(\./)*}, '')
  end
end
