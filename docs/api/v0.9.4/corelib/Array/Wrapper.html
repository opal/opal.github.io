def [](index, length = undefined)
  %x{
    var result = #{@literal.slice(index, length)};

    if (result.$$is_array && (index.$$is_range || length !== undefined)) {
      return #{self.class.allocate(`result`)}
    }
    else {
      return result;
    }
  }
