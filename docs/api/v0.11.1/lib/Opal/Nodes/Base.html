def closest_module_node
  current = scope
  while current && !current.class_scope?
    current = current.parent
  end
  current
end

    str = fragment(str) if str.is_a?(String)
    @fragments << str
  end

def unshift(*strs)
  strs.reverse.each do |str|
    str = fragment(str) if str.is_a?(String)
    @fragments.unshift str
  end
end
