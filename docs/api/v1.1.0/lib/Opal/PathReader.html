def expand(path)
  if Pathname.new(path).absolute? || path =~ RELATIVE_PATH_REGEXP
    path
  else
    find_path(path)
  end
end

def read(path)
  full_path = expand(path)
  return nil if full_path.nil?
  File.open(full_path, 'rb:UTF-8', &:read) if File.exist?(full_path)
end
