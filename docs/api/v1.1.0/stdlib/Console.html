def group(*args, &block)
  raise ArgumentError, 'no block given' unless block

  `#{@native}.group.apply(#{@native}, args)`

  begin
    if block.arity == 0
      instance_exec(&block)
    else
      yield(self)
    end
  ensure
    `#{@native}.groupEnd()`
  end
end
