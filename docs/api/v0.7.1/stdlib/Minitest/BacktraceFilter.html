def filter bt
  return ["No backtrace"] unless bt

  return bt.dup if $DEBUG

  new_bt = bt.take_while { |line| line !~ /lib\/minitest/ }
  new_bt = bt.select     { |line| line !~ /lib\/minitest/ } if new_bt.empty?
  new_bt = bt.dup                                           if new_bt.empty?

  new_bt
end
