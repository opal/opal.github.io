def implementation?(*args)
  args.any? do |name|
    !!case name
    when :rubinius
      RUBY_NAME =~ /^rbx/
    when :ruby
      RUBY_NAME =~ /^ruby/
    when :jruby
      RUBY_NAME =~ /^jruby/
    when :ironruby
      RUBY_NAME =~ /^ironruby/
    when :macruby
      RUBY_NAME =~ /^macruby/
    when :maglev
      RUBY_NAME =~ /^maglev/
    when :topaz
      RUBY_NAME =~ /^topaz/
    when :opal
      RUBY_NAME =~ /^opal/
    else
      false
    end
  end
end
