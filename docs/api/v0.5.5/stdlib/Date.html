def ==(other)
  %x{
    var a = #{self}._date, b = #{other}._date;
    a.setHours(0, 0, 0, 0);
    b.setHours(0, 0, 0, 0);
    return (a.getFullYear() === b.getFullYear() && a.getMonth() === b.getMonth() && a.getDate() === b.getDate());
  }
end
