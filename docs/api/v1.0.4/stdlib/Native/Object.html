def []=(key, value)
  native = ::Native.try_convert(value)

  if `#{native} === nil`
    `#{@native}[key] = #{value}`
  else
    `#{@native}[key] = #{native}`
  end
end

def each(*args)
  if block_given?
    %x{
      for (var key in #{@native}) {
        #{yield `key`, `#{@native}[key]`}
      }
    }

    self
  else
    method_missing(:each, *args)
  end
end

def merge!(other)
  %x{
    other = #{::Native.convert(other)};

    for (var prop in other) {
      #{@native}[prop] = other[prop];
    }
  }

  self
end
