def initialize(args)
  @args = []
  @optargs = []
  @restarg = nil
  @postargs = []
  @kwargs = []
  @kwoptargs = []
  @kwrestarg = nil
  @shadowargs = []
  @blockarg = nil

  args.each do |arg|
    case arg.type
    when :arg, :mlhs
      (@restarg || @optargs.any? ? @postargs : @args) << arg
    when :optarg
      @optargs << arg
    when :restarg
      @restarg = arg
    when :kwarg
      @kwargs << arg
    when :kwoptarg
      @kwoptargs << arg
    when :kwrestarg
      @kwrestarg = arg
    when :shadowarg
      @shadowargs << arg
    when :blockarg
      @blockarg = arg
    else
      raise "Unsupported arg type #{arg.type}"
    end
  end
end

def can_inline_kwargs?
  @optargs.empty? && @restarg.nil? && @postargs.empty?
end

def has_post_args?
