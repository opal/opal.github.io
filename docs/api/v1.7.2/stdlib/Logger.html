def add(severity, message = nil, progname = nil, &block)
  return true if (severity ||= UNKNOWN) < @level
  progname ||= @progname
  unless message
    if block_given?
      message = yield
    else
      message = progname
      progname = @progname
    end
  end
  @pipe.write(@formatter.call(SEVERITY_LABELS[severity] || 'ANY', ::Time.now, progname, message))
  true
end
