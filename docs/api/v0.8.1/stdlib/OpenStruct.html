def initialize(hash = nil)
  @table = {}

  hash.each_pair {|key, value|
    @table[key.to_sym] = value
  } if hash
end

def method_missing(name, *args)
  if name.end_with? '='
    @table[name[0 .. -2].to_sym] = args[0]
  else
    @table[name.to_sym]
  end
end
