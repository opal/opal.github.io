def /(other)
  case other
  when Numeric
    Scalar.new(@value / other)
  when Vector
    Scalar.Raise ErrOperationNotDefined, "/", @value.class, other.class
  when Matrix
    self * other.inverse
  else
    apply_through_coercion(other, __method__)
  end
end
