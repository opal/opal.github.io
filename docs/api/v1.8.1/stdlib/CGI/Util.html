def unescapeHTML(string)
  string.gsub(/&(apos|amp|quot|gt|lt|\#[0-9]+|\#[xX][0-9A-Fa-f]+);/) do
    match = ::Regexp.last_match(1)
    case match
    when 'apos'                then "'"
    when 'amp'                 then '&'
    when 'quot'                then '"'
    when 'gt'                  then '>'
    when 'lt'                  then '<'
    when /\A#0*(\d+)\z/
      n = ::Regexp.last_match(1).to_i
      n.chr('utf-8')
    when /\A#x([0-9a-f]+)\z/i
      n = ::Regexp.last_match(1).hex
      n.chr('utf-8')
    else
      "&#{match};"
    end
  end
end
