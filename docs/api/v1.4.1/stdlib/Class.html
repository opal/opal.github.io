def native_alias(new_jsid, existing_mid)
  %x{
    var aliased = #{self}.prototype['$' + #{existing_mid}];
    if (!aliased) {
      #{raise NameError.new("undefined method `#{existing_mid}' for class `#{inspect}'", existing_mid)};
    }
    #{self}.prototype[#{new_jsid}] = aliased;
  }
end
