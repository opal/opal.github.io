def initialize(str)
  @str = str

  str =~ /^(.*?):(\d+):(\d+):in `(.*?)'$/
  @path = Regexp.last_match(1)
  @label = Regexp.last_match(4)
  @lineno = Regexp.last_match(2).to_i

  @label =~ /(\w+)$/
  @base_label = Regexp.last_match(1) || @label
end
