def on_kwsplat(node)
  hash, _ = *node

  if hash.type == :hash
    hash = process_regular_node(hash)
  end

  node.updated(nil, [hash])
end

def on_pair(node)
  key, _value = *node

  if %i[str sym].include?(key.type)
    @keys << key
  end

  super(node)
end
