def gen_args(arg_count)
  (1..arg_count).map do |i|
    s(:arg, :"_#{i}")
  end
end

def on_numblock(node)
  left, arg_count, right = node.children

  s(
    :block,
    left,
    s(:args, *gen_args(arg_count)),
    right
  )
end
