  elsif simple_keys?
    compile_hash2
  else
    compile_hash
  end
end

def compile_hash
  helper :hash

  children.each_with_index do |child, idx|
    push ', ' unless idx == 0
    push expr(child)
  end

  wrap '$hash(', ')'
end

def compile_hash2
  hash_obj, hash_keys = {}, []
  helper :hash2

  keys.size.times do |idx|
    key = keys[idx][1].to_s.inspect
    hash_keys << key unless hash_obj.include? key
    hash_obj[key] = expr(values[idx])
  end

  hash_keys.each_with_index do |key, idx|
    push ', ' unless idx == 0
    push "#{key}: "
    push hash_obj[key]
  end

  wrap "$hash2([#{hash_keys.join ', '}], {", "})"
end

def compile_merge
  helper :hash

  result, seq = [], []

  children.each do |child|
    if child.type == :kwsplat
      unless seq.empty?
        result << expr(s(:hash, *seq))
      end
      result << expr(child)
      seq = []
    else
      seq << child
    end
  end
  unless seq.empty?
    result << expr(s(:hash, *seq))
  end

  result.each_with_index do |fragment, idx|
    if idx == 0
      push fragment
    else
      push ".$merge(", fragment, ")"
    end
  end
end

def extract_kv_pairs_and_kwsplats
  found_key = false

  children.each do |obj|
    if obj.type == :kwsplat
      self.has_kwsplat = true
    elsif found_key
      values << obj
      found_key = false
    else
      keys << obj
      found_key = true
    end
  end

  [keys, values]
end
