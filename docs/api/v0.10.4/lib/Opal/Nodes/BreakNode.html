def break_val
  if value.nil?
    expr(s(:nil))
  elsif children.size > 1
    expr(s(:array, *children))
  else
    expr(value)
  end
end

def compile
  if in_while?
    compile_while
  elsif scope.iter?
    compile_iter
  else
    error "void value expression: cannot use break outside of iter/while"
  end
end

def compile_iter
